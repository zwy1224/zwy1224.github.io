03-HelloWorld代码解释
//public 全局的定义
//class 表示定义一个类
//HelloWorld 表示我们的类名，定义的时候最好见名知意，最好首字母都要大写
public class HelloWorld
{
//static 表示静态
//void 表示没有返回值
//main 方法名，表示程序的入口
//括号里面的内容表示方法的参数
public static void main(String[]args){
//就是在控制台上打印HelloWorld!
System.out.println(“HelloWorld!”);
}

}
